## Usage

```tsx
import { DropdownMenu, Button } from "@easy-shadcn/react"

<DropdownMenu
  menu={[
    {
      name: 'Profile',
      key: 'profile',
      shortcut: '⇧⌘P',
    },
    {
      name: 'Log Out',
      key: 'logout',
      shortcut: '⇧⌘Q',
    },
    {
      groupName: 'More',
      key: 'more',
      items: [
        {
          prefix: 'PREFIX',
          name: 'Github',
          key: 'github',
        },
        {
          name: 'Support',
          key: 'support',
        },
        {
          name: 'API',
          key: 'api',
          disabled: true
        },
      ],
    },
  ]}
>
  <Button>Here</Button>
</DropdownMenu>
```
