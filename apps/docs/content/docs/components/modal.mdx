## Modal Usage

```tsx
import { Modal } from "@easy-shadcn/react"

<Modal
  title='Modal title'
  description='Modal description'
  content={(
    <div>
      <div>Modal Content1</div>
      <div>Modal Content2</div>
      <div>Modal Content3</div>
    </div>
  )}
>
  <Button>Click Open</Button>
</Modal>
```

## Modal Action Usage

Add ModalHost to APP root

```tsx
import { ModalHost } from "@easy-shadcn/react"

const App = () => {
  <div>
    {/* Other */}
    {ModalHost}
  </div>
}
```

```tsx
import { modalAction } from "@easy-shadcn/react"

modalAction.confirm({
  title: 'Tips',
  content: 'Hi here',
  onCancel: () => {
    console.log('cancel')
  },
  onConfirm: () => {
    console.log('confirm')
  })
})

modalAction.open({
  title: 'title',
  content: 'Modal Content',
  footer: 'Modal Footer'
})
```
