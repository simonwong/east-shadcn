## Alert Modal Usage

Add AlertModalHost to APP root

```tsx
import { AlertModalHost } from "@easy-shadcn/react"

const App = () => {
  <div>
    {/* Other */}
    {AlertModalHost}
  </div>
}
```

```tsx
import { alertModalAction } from "@easy-shadcn/react"

alertModalAction.confirm({
  title: 'Tips',
  content: 'Hi here',
  onCancel: () => {
    console.log('cancel')
  },
  onConfirm: () => {
    console.log('confirm')
  })
})
```
