## Usage

```tsx
import { Toggle } from "@easy-shadcn/react"

<Toggle
  type='single'
  options={[
    {
      value: 'first',
      label: "First"
    },
    {
      value: 'second',
      label: "Second"
    },
  ]}
/>
```
