Toast is implemented based on [sooner](https://sonner.emilkowal.ski/)

## Usage

```tsx
import { Toaster, toast } from "@easy-shadcn/react"

// Root
<div>
  <Toaster />
</div>

// use

toast.success('success')
```

## Examples

### Basic

<CodeDemo code={`
const Demo = () => {
  return (
    <div>
      <Button
        onClick={() => {
          toast.success('Hi Here, success', {
            duration: Infinity,
            closeButton: true,
          })
        }}
      >
        success
      </Button>
      <Button
        onClick={() => {
          toast.error('Hi Here, error')
        }}
      >
        error
      </Button>
      <Button
        onClick={() => {
          toast.info('Hi Here, info')
        }}
      >
        info
      </Button>
      <Toaster richColors />
    </div>
  )
}
render(<Demo />)
`}
/>
