## Usage

```tsx
import { Tabs } from "@easy-shadcn/react"

<Tabs
  defaultValue="first"
  option={[
    {
      title: 'First',
      key: 'first',
      content: (
        <div>First Content</div>
      )
    },
    {
      title: 'Second',
      key: 'Second',
      content: (
        <div>Second Content</div>
      )
    },
  ]}
/>
```
